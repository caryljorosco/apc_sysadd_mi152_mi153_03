//CUSTOMER
CREATE TABLE CUSTOMER(ID INT NOT NULL PRIMARY KEY AUTO_INCREMENT, Last_Name VARCHAR(45) NOT NULL, First_Name VARCHAR(45) NOT NULL, Middle_Name VARCHAR(45) NOT NULL, Contact_Number VARCHAR(15) NOT NULL);

//ROOMSCHEDULE
CREATE TABLE ROOMSCHEDULE(ID INT NOT NULL PRIMARY KEY AUTO_INCREMENT, Date DATE NOT NULL, Room_Number VARCHAR(2) NOT NULL, Time TIME NOT NULL);

//EMPLOYEE
CREATE TABLE EMPLOYEE(ID INT NOT NULL PRIMARY KEY, Last_Name VARCHAR(45) NOT NULL, First_Name VARCHAR(45) NOT NULL, Middle_Name VARCHAR(45) NOT NULL, Position VARCHAR(45) NOT NULL);

//SERVICES
CREATE TABLE SERVICES(ID INT NOT NULL PRIMARY KEY AUTO_INCREMENT, Service_Name VARCHAR(45) NOT NULL, Description VARCHAR(200), Duration VARCHAR(45) NOT NULL, Price DECIMAL NOT NULL);

//BOOKING
CREATE TABLE BOOKING(ID INT NOT NULL PRIMARY KEY AUTO_INCREMENT, Booking_Type VARCHAR(15) NOT NULL, Date_Time_Received DATETIME NOT NULL, Customer_ID INT, Employee_ID INT, FOREIGN KEY (Customer_ID) REFERENCES customer(ID), FOREIGN KEY (Employee_ID) REFERENCES employee(ID));

//BOOKING_DETAILS
CREATE TABLE booking_details(ID INT NOT NULL PRIMARY KEY AUTO_INCREMENT, Booking_ID INT, Employee_ID INT, Services_ID INT, RoomSchedule_ID INT, FOREIGN KEY (Booking_ID) REFERENCES booking(ID), FOREIGN KEY (Employee_ID) REFERENCES employee(ID), FOREIGN KEY (Services_ID) REFERENCES services(ID), FOREIGN KEY (RoomSchedule_ID) REFERENCES roomschedule(ID));
